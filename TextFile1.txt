Need a table to store entries, application table, application ID, companyName, job title, etc. 

CREATE TABLE Applications (
  ApplicationId INT IDENTITY(1,1) PRIMARY KEY,
  CompanyName   NVARCHAR(200) NOT NULL,
  JobTitle      NVARCHAR(200) NULL,
);

Need one to store the stages, stageID, stageKey, DisplayName, SortOrder

CREATE TABLE Stages (
  StageId     INT IDENTITY(1,1) PRIMARY KEY,
  StageKey    VARCHAR(40) UNIQUE NOT NULL,  -- APPLIED, INT1, INT2, INT3_5, OFFER, HIRED, REJECTED, NO_RESPONSE
  DisplayName VARCHAR(100) NOT NULL,
  SortOrder   INT NOT NULL
);

one to many relationship table between the two (one application can have many stages)

CREATE TABLE ApplicationEvents (
  EventId       BIGINT IDENTITY(1,1) PRIMARY KEY,
  ApplicationId INT NOT NULL FOREIGN KEY REFERENCES Applications(ApplicationId) ON DELETE CASCADE,
  StageId       INT NOT NULL  FOREIGN KEY REFERENCES Stages(StageId),
);
